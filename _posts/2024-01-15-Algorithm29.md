## 201.   Happy Number


```c++
class Solution {
public:
    bool isHappy(int n) {
        vector<int> trackNum;
        int target = n;
        while(true)
        {
            int sum = 0;
            vector<int> digits;
            while(target != 0)
            {
                digits.push_back((target%10) * (target%10));
                target = target/10;
            }
            for(int i =0; i<digits.size(); i++){
                sum += digits[i];
            }
            if(sum == 1)
                break;
            if(find(trackNum.begin(), trackNum.end(), sum) == trackNum.end())
            {
                trackNum.push_back(sum);
            }else
            {
                return false;
            }
            target = sum;
        }

        return true;
    }
};
```
