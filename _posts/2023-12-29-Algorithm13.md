## 191. Number of 1 Bits


```c++
class Solution {
public:
    int hammingWeight(uint32_t n) {
        int count = 0;
        if(n == 0)
            return count;
        while(n!=1){
            if(n%2==1)
                count++;
            n = n/2;
        }
        return count+1;
    }
};
```
